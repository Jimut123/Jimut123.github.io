1. Modern C++ Smart Pointers

Using raw pointers often leads to memory leaks. Modern C++ introduced smart pointers via the <memory> header to manage object lifecycles automatically.


#include <iostream>
#include <memory>
#include <vector>

class Processor {
public:
    Processor() { std::cout << "Processor Created\n"; }
    ~Processor() { std::cout << "Processor Destroyed\n"; }
    void run() { std::cout << "Running calculations...\n"; }
};

int main() {
    // std::unique_ptr ensures single ownership
    std::unique_ptr<Processor> ptr = std::make_unique<Processor>();
    ptr->run();

    // No delete needed. Memory is freed when 'ptr' goes out of scope.
    return 0;
}
