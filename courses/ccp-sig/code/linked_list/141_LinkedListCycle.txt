# Definition for singly-linked list.
# class ListNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution(object):
    def hasCycle(self, head):
        """
        :type head: ListNode
        :rtype: bool
        """
        if head is None:
            return False
        head1=head;head2=head
        while head2.next is not None and head2.next.next is not None:
            head1=head1.next
            head2=head2.next.next
            if head1==head2:
                return True
        return False
