class Solution(object):
    def twoSum(self, numbers, target):
        """
        :type numbers: List[int]
        :type target: int
        :rtype: List[int]
        """
        L=0;R=len(numbers)-1
        while L<R:
            if numbers[L]+numbers[R]<target:
                L+=1
            elif numbers[L]+numbers[R]>target:
                R-=1
            else:
                return [L+1,R+1]
        return 0
