class NumArray(object):

    def __init__(self, nums):
        """
        :type nums: List[int]
        """
        self.prefix_sum = []
        sum=0
        for i in nums:
            sum+=i
            self.prefix_sum.append(sum)
        

    def sumRange(self, left, right):
        """
        :type left: int
        :type right: int
        :rtype: int
        """
        if left>0: 
            return self.prefix_sum[right]-self.prefix_sum[left-1]
        return self.prefix_sum[right]
