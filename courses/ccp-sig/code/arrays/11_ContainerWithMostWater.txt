class Solution(object):
    def maxArea(self, height):
        """
        :type height: List[int]
        :rtype: int
        """
        L=0;R=len(height)-1
        max_vol=0
        while L<R:
            max_vol = max(max_vol, (R-L)*min(height[L],height[R]))
            if height[L]<height[R]:
                L+=1
            else:
                R-=1
        return max_vol
