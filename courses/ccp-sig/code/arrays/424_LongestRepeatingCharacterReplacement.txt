class Solution(object):
    def characterReplacement(self, s, k):
        """
        :type s: str
        :type k: int
        :rtype: int
        """
        longest=0
        L=0
        counts=[0]*26
        for R in range(len(s)):
            counts[ord(s[R])-65]+=1
            while (R-L+1)-max(counts)>k:
                counts[ord(s[L])-65] -=1
                L+=1
            longest=max(longest,R-L+1)
        return longest
